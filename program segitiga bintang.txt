program Segitiga_bintang;
uses crt;
var
i,j,n,x: integer;

begin
writeln('Segitiga Bintang Rata tengah');
write('Masukan jumlah baris : ');readln(n);
writeln;
for i:=1 to n do
begin
  for j:=1 to n-i do
  begin
  write(' ');
  end;

  for x:=i downto 1 do
  begin
  write('* ');
  end;

writeln;
end;

readkey;
end.
